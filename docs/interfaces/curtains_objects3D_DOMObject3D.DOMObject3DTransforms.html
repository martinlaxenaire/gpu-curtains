Interface DOMObject3DTransforms

Defines all necessary vectors/quaternions to compute a 3D model matrix based on a DOM HTMLElement

interface DOMObject3DTransforms {
    quaternion: Quat;
    rotation: Vec3;
    scale: Vec3;
    origin: { model: Vec3; world: Vec3 };
    position: { world: Vec3; document: Vec3 };
}
